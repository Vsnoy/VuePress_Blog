# git log 简洁模式

使用别名自定义 git logp 指令，简化 log 输出日志

git config --global alias.logp 'log --pretty=format:%cd:%cn:%h:%s'
1
git config --list|grep alias
alias.st=status
alias.ck=checkout
alias.br=branch
alias.cf=config
alias.fc=fetch
alias.brv=branch --v
alias.geturl=config --get remote.origin.url
alias.logp=log --pretty=format:%cd:%cn:%h:%s
